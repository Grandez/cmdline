133 string Argument::applyDefault(const char* txt) {
134 char aux[256] = "";
135 if (txt != NULL && strlen(txt) > 0) return string(txt);
136 std::time_t t = std::time(0);
137 std::tm* now = std::localtime(&t);
138 switch (this->type) {
139 case Type::DATE:
140 defaultDate(aux);
141 break;
142 case Type::TIME:
143 sprintf(aux, "%02d:%02d:%02d", now->tm_hour, now->tm_min, now->tm_sec);
144 break;
145 case Type::DATETIME:
146 sprintf(aux, "%04d-%02d-%02d %02d:%02d:%02d",
147 now->tm_year + 1900, now->tm_mon + 1, now->tm_mday,
148 now->tm_hour, now->tm_min, now->tm_sec);
149 break;
150 case Type::TMS:
151 sprintf(aux, "%04d-%02d-%02d-%02d:%02d:%02d.000000",
152 now->tm_year + 1900, now->tm_mon + 1, now->tm_mday,
153 now->tm_hour, now->tm_min, now->tm_sec);
154 break;
155 case Type::DIR:
156 case Type::DIR_EXISTS:
157 strcpy(aux, (const char *) std::filesystem::current_path().string().c_str());
158 break;
159 case Type::BOOL:
160 aux[0] = '0'; aux[1] = 0x0;
161 break;
162 default:
163 if (txt == NULL) return string("");
