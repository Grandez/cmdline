35 struct tm* makeDateTime(char* sdate, char* stime) {
36 struct tm* tt;
37 struct tm now;
38 struct tm res;
39
40 time_t t = std::time(0);
41 tt = std::localtime(&t);
42 memcpy(&now, tt, sizeof(tm));
43
44 if (!sdate) {
45 memcpy(&res, &now, sizeof(tm));
46 }
47 else {
48 std::istringstream ss(sdate);
49 ss >> get_time(&res, "%Y/%m/%d");
50 res.tm_hour = now.tm_hour;
51 res.tm_min = now.tm_min;
52 res.tm_sec = now.tm_sec;
53 }
54
55 if (stime) {
56 vector<int> tt = tokenizeNumber(stime, ":");
57 res.tm_hour = tt[0];
58 res.tm_min = tt[1];
59 res.tm_sec = tt[2];
60 }
61 tt = (struct tm *) malloc(sizeof(struct tm));
62 memcpy(tt, &res, sizeof(struct tm));
63 return tt;
64 }

177 struct tm* validateTime (const char* value) {
178 char strTime[9];
179 regex pat{ "^[0-9]{1,2}:[0-9]{1,2}:[0-9]{1,2}$" };
180 bool match = regex_search(value, pat);
181 if (!match) throw CmdLineValueException(TXT_VAL_EXPECTED, value, TXT_TIME);
182 vector<int> res = tokenizeNumber(value, (char*)":");
183 if (res[0] < 0 || res[0] > 23) throw CmdLineValueException(TXT_VAL_INVALID, value);
184 for (int i = 1; i < 3; i++)
185 if (res[i] < 0 || res[i] > 59) throw CmdLineValueException(TXT_VAL_INVALID, value);
186 sprintf(strTime, "%02d:%02d:%02d", res[0], res[1], res[2]);
187 return makeDateTime(0x0, strTime);
188 }
189 void validateDateValue (const char *value, vector<int> dt) {
190 int days[] = { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };

198 struct tm* validateDate (const char* value, int fmt) {
199 char strDate[11];
200 time_base::dateorder d;
201 vector<int> dt(3);
202 regex pat1{ "^[0-9]{1,2}[/-]{1}[0-9]{1,2}[/-]{1}[0-9]{2,4}$" };
203 regex pat2{ "^[0-9]{2,4}[/-]{1}[0-9]{1,2}[/-]{1}[0-9]{1,2}$" };
204 regex pat;
205 d = (fmt == -1) ? use_facet<time_get<char>>(locale()).date_order()
206 : static_cast<time_base::dateorder>(fmt);
207
208 switch (d) {
209 case time_base::no_order:
210 case time_base::dmy:
211 case time_base::mdy: pat = pat1; break;
212 default: pat = pat2;
213 }
214 bool match = regex_search(value, pat);
215 if (!match) throw CmdLineValueException(TXT_VAL_EXPECTED, value, TXT_DATE);
216 vector<int> res = tokenizeNumber(value, (char*)"[/-]");
217 switch (d) {
218 case time_base::dmy: dt = { res[0], res[1], res[2] }; break;
219 case time_base::mdy: dt = { res[1], res[0], res[2] }; break;
220 case time_base::ymd: dt = { res[2], res[1], res[0] }; break;
221 case time_base::ydm: dt = { res[2], res[0], res[1] }; break;
222 default: dt = { res[0], res[1], res[2] };
223 }
224 validateDateValue(value, dt);
225 sprintf(strDate, "%04d/%02d/%02d", dt[2], dt[1], dt[0]);
226 return makeDateTime(strDate, 0x0);
227 }
228 struct tm* validateDateTime (const char* value) {
229 struct tm* tmDate;
230 struct tm* tmTime;
231 char szDate[11];
232 char szTime[10];
233 vector<int> res;
234 // Revisited. More clean by pieces
235 vector<string> pieces = tokenize(value, "[ \t]+");
236 if (pieces.size() != 2) throw CmdLineValueException(TXT_VAL_EXPECTED, value, TXT_DATETIME);
237 try {
238 tmDate = validateDate(pieces[0].c_str(), time_base::ymd);
239 tmTime = validateTime(pieces[1].c_str());
240 sprintf(szDate, "%04d/%02d/%02d", tmDate->tm_year + 1900, tmDate->tm_mon + 1, tmDate->tm_mday);
241 sprintf(szTime, "%02d:%02d:%02d", tmTime->tm_hour, tmTime->tm_min, tmTime->tm_sec);
242 }
243 catch(exception ex) {
244 throw CmdLineValueException(TXT_VAL_INVALID, value);
245 }
246 return makeDateTime(szDate,szTime);

248 char * validateTimestamp (const char* value) {
249 vector<int> res;
250 // timestamp es yyyy-mm-dd[- ]hh:MM:SS.nnnnnn
251 struct tm *tmDate, *tmTime;
252 vector<string> pieces = tokenize(value, "-|[ \t]+");
253 int npieces = (int) pieces.size();
254 if (npieces != 2 && npieces != 4) throw CmdLineValueException(TXT_VAL_EXPECTED, value, TXT_TMS);
255 try {
256 int iTime = (npieces == 2) ? 1 : 3;
257 if (npieces == 2) tmDate = validateDate(pieces[0].c_str(), time_base::ymd);
258 if (npieces != 2) {
259 char strdt[11];
260 sprintf(strdt, "%s-%s-%s", pieces[0].c_str(), pieces[1].c_str(), pieces[2].c_str());
261 tmDate = validateDate(strdt, time_base::ymd);
262 }
263 vector<string> tt = tokenize(pieces[iTime].c_str(), "\\.");
264 if (tt.size() != 2) throw CmdLineValueException(TXT_VAL_EXPECTED, value, TXT_TMS);
265 tmTime = validateTime(tt[0].c_str());
266 validateNumber(tt[1].c_str());
267 }
268 catch (exception ex) {
269 throw CmdLineValueException(TXT_VAL_INVALID, value);
270 }
271 return (char *) value;

273 filesystem::path validateDir (const char* value) {
274 // If is not a valid path, chdir return -1
275 // Some times chdir returns 0 but not change the directory
276 char old[256];
277 char tmp[256];
278 int rc, rc2;
279 char* ptr = getcwd(old, 256);
280 if (strcmp(value, ".") == 0) return filesystem::path(old);
281 if (strcmp(value, old) == 0) return filesystem::path(old);
282 rc = _chdir(value);
283 if (rc) throw CmdLineValueException(TXT_VAL_EXPECTED, value, TXT_DIR );
284 ptr = getcwd(tmp, 256);
285 rc = chdir(old);
286 #ifdef _WIN32 // Windows is case insensitive
287 rc2 = strcmpi(old, tmp);
288 #else
289 rc2 = strcmp(old, tmp);
290 #endif
291
292 if (rc2 == 0) throw CmdLineValueException(TXT_VAL_EXPECTED, value, TXT_DIR);
293 return filesystem::path(value);
