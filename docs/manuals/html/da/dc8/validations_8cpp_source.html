169 char strTime[9];
170 regex pat{ "^[0-9]{1,2}:[0-9]{1,2}:[0-9]{1,2}$" };
171 bool match = regex_search(value, pat);
172 if (!match) throw CmdLineValueException(TXT_VAL_EXPECTED, value, TXT_TIME);
173 vector<int> res = tokenizeNumber(value, (char*)":");
174 if (res[0] < 0 || res[0] > 23) throw CmdLineValueException(TXT_VAL_INVALID, value);
