143 string Argument::applyDefault(const char* txt) {
144 char aux[256] = "";
145 if (txt != NULL && strlen(txt) > 0) return string(txt);
146 std::time_t t = std::time(0);
147 std::tm* now = std::localtime(&t);
148 switch (this->type) {
149 case Type::DATE:
150 defaultDate(aux);
151 break;
152 case Type::TIME:
153 sprintf(aux, "%02d:%02d:%02d", now->tm_hour, now->tm_min, now->tm_sec);
154 break;
155 case Type::DATETIME:
156 sprintf(aux, "%04d-%02d-%02d %02d:%02d:%02d",
157 now->tm_year + 1900, now->tm_mon + 1, now->tm_mday,
158 now->tm_hour, now->tm_min, now->tm_sec);
159 break;
160 case Type::TMS:
161 sprintf(aux, "%04d-%02d-%02d-%02d:%02d:%02d.000000",
162 now->tm_year + 1900, now->tm_mon + 1, now->tm_mday,
163 now->tm_hour, now->tm_min, now->tm_sec);
164 break;
165 case Type::DIR:
166 case Type::DIR_EXISTS:
167 strcpy(aux, (const char *) std::filesystem::current_path().string().c_str());
168 break;
169 case Type::BOOL:
170 aux[0] = '0'; aux[1] = 0x0;
171 break;
172 default:
173 if (txt == NULL) return string("");
174 return string(aux);
175 }
176 return (string(aux));
177 }
