103 void _CommandLine::parse(int argc, char** argv) {
104 char c;
105 char szOpt[64] = "";
106 std::string in;
107 char* prev = nullptr;
108 for (int i = 1; i < argc; i++) {
109 c = argv[i][0];
110
111 if (c == '/' || c == '+' || c == '-') {
112 strcpy(szOpt, argv[i]);
113 if (sensitive) {
114 for (size_t j = 0; j < strlen(szOpt); j++) szOpt[j] = toupper(szOpt[j]);
115 }
116 }
117 switch (argv[i][0]) {
118 case '/':
119 prev = updateOption(szOpt, prev);
120 if (strict) checkAlreadySet(&options, prev);
121 break;
122 case '+': if (prev != nullptr) throw CmdLineException(ERR_ARG_MISSING, prev);
123 prev = updateFlag(szOpt, argv[i], true); break;
124 case '-': in = string(szOpt);
125 prev = (in == std::string("-h") || in == std::string("--help")) ?
126 updateFlag((char*)"+help", prev, true) :
127 updateFlag(argv[i], prev, false);
128 break;
129 default:
130 if (prev == NULL) inputs.push_back(argv[i]);
131 if (prev != NULL) prev = addValueToOption(argv[i], prev);
132 }
133 }
134 if (hasFlag("HELP")) throw HelpDetailedRequested();
135 if (hasFlag("help")) throw HelpRequested();
136 }

140 char* _CommandLine::checkParameter (ParameterTree* root[], const char* parm) {
141 size_t idx = 0;
142 ParameterTree* base = root[parm[0] - ' '];
143 ParameterTree* prev = nullptr;
144
145 while (idx < strlen(parm) && base != nullptr) {
146 prev = base;
147 if (base->letter != parm[idx]) break;
148 base = base->getChild(parm[++idx]);
149 }
150
151 std::string str;
152 if (idx == strlen(parm)) { // Parsing full done
153 switch (prev->numChildren()) {
154 case 0: return (strdup(parm));
155 case 1: str = parm;
156 str.append(prev->getNext()->getWord());
157 return (makeChar(str));
158 default: throw CmdLineParameterException(ERR_ARG_ERR, parm);
159 }
160 }
161 if (prev == nullptr) throw CmdLineException(TXT_ARG_NFND, parm); // primero
162 if (idx < strlen(parm) && prev->branchs == 0) { // overtype
163 throw CmdLineException(TXT_ARG_NFND1, parm, prev->getReversedWord());
164 }
165 // Ha salido por base = nullptr
166
167 if (prev->branchs == 1) {
168 char * next = prev->getWord();
169 char* base = prev->getReversedWord();
170 base[strlen(base) - 1] = 0x0;
171 throw CmdLineException(TXT_ARG_NFND2, parm, base, next);
172 }
173 if (prev->branchs > 1) {
174 throw CmdLineException(TXT_ARG_NFND3, parm);
175 }
176// if (base == nullptr) throw CmdLineParameterException(parm, prev->getReversedWord());
177
178 std::string ss(parm);
179 ss.append(base->getWord());
180 return (makeChar(ss));
181 }

209 char* _CommandLine::updateDefinition (const char* value) {
210 const char* ptr = &(value[1]);
211 vector<string> toks = tokenize(ptr, "=");
212 if (toks.size() != 2) throw CmdLineParameterException(INV_DEFINITION, value);
213 if (toks[0].length() == 0) throw CmdLineParameterException(INV_DEFINITION, value);
214 Argument *def = defines.find(toks[0]);
215 if (def == nullptr) def = new Define(toks[0].c_str());
216 def->initValues(splitArgument(toks[1].c_str()));

218 return nullptr;
